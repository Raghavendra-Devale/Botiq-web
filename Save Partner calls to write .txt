async deleteJobOrder(jobId: number): Promise<{ success: boolean }> {
  if (!this.db) {
    console.error('Database not initialized');
    return { success: false };
  }

  try {
    const docCountResult = await this.db.query(
      `SELECT COUNT(*) as count FROM botiq_job_docs WHERE job_id = ?`,
      [jobId]
    );
    const docCount = docCountResult.values?.[0]?.count || 0;

    if (docCount > 0) {
      const docIdResult = await this.db.query(
        `SELECT job_doc_id FROM botiq_job_docs WHERE job_id = ?`,
        [jobId]
      );
      const jobDocIds = docIdResult.values?.map((row: any) => row.job_doc_id) || [];

      await this.db.query(`DELETE FROM botiq_job_docs WHERE job_id = ?`, [jobId]);

      for (const docId of jobDocIds) {
        await this.logSyncDelete('botiq_job_docs', 'job_doc_id', docId);
        console.log(` Logged deletion for job_doc_id: ${docId}`);
      }

      console.log(` Deleted ${jobDocIds.length} document(s) for job_id: ${jobId}`);
    }

    const jobOrderResult = await this.db.query(
      `SELECT COUNT(*) as count FROM botiq_job_order WHERE job_id = ?`,
      [jobId]
    );
    const jobExists = jobOrderResult.values?.[0]?.count > 0;

    if (!jobExists) {
      console.warn(` No job order found with job_id: ${jobId}`);
      return { success: false };
    }

    // Step 6: Delete the job order
    await this.db.query(`DELETE FROM botiq_job_order WHERE job_id = ?`, [jobId]);
    await this.logSyncDelete('botiq_job_order', 'job_id', jobId);
    console.log(` Job order deleted successfully for job_id: ${jobId}`);

    return { success: true };
  } catch (error) {
    console.error(' Error deleting job order and its documents:', error);
    return { success: false };
  }
}




async getCategoriesByPartnerId(partnerId: number): Promise<any[]> {
  try {
    // Fetch partner category IDs and names for the given partner
    const query = `SELECT partner_category_id, partner_category FROM botiq_partner WHERE partner_id = ?`;
    const result = await this.db.query(query, [partnerId]);

    if (result.values.length > 0) {
      const partnerCategoryIds = result.values[0].partner_category_id;
      const partnerCategories = result.values[0].partner_category;

      if (partnerCategoryIds && partnerCategories) {
        const categoryIdsArray = partnerCategoryIds.split(',').map((id:any) => id.trim());
        const categoryNamesArray = partnerCategories.split(',').map((name:any) => name.trim());

        // Combine IDs and Names into an array of objects
        return categoryIdsArray.map((id:any, index:any) => ({
          id: id,
          name: categoryNamesArray[index] || 'Unknown', // Ensure names match with IDs
        }));
      }
    }

    return []; // Return empty array if no categories found
  } catch (error) {
    console.error(`Error fetching categories for partner ${partnerId}:`, error);
    return [];
  }
}

 async saveOrUpdateJobOrders(jobOrder: any): Promise<number | null> {
    if (!this.db) {
      console.error(' Database not initialized');
      return null;
    }
  
    const currentDate = this.getLocalDateTime()
  
    try {
      //  Update Case
      if (jobOrder.job_id) {
        const updateQuery = `
          UPDATE botiq_job_order 
          SET org_id = ?, order_id = ?, customer_id = ?, partner_id = ?, 
              job_order_details = ?, job_due_date = ?, job_priority = ?,

          WHERE job_id = ?;
        `;

          jobOrder.order_id,
          jobOrder.customer_id,
          jobOrder.partner_id,
          jobOrder.job_order_details,
          jobOrder.job_due_date,
          jobOrder.job_priority,
          jobOrder.job_order_status,
          currentDate,
          jobOrder.updated_by || null,
          jobOrder.job_id
        ];
  
        const result = await this.db.run(updateQuery, updateValues);
  
        if (result.changes?.changes > 0) {
          await this.logSyncChange('botiq_job_order', 'job_id', jobOrder.job_id);
          console.log(' Job order updated with ID:', jobOrder.job_id);
        } else {
          console.warn(` No update performed. Possibly job_id ${jobOrder.job_id} not found.`);
        }
  
        return jobOrder.job_id;
      }
  
      //  Insert Case
      const insertQuery = `
        INSERT INTO botiq_job_order (
          org_id, order_id, customer_id, partner_id, job_order_details,
          job_due_date, job_priority, job_order_status,
          created_date, updated_date, updated_by
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?);
      `;
  
      const insertValues = [
        jobOrder.org_id,
        jobOrder.order_id,
        jobOrder.customer_id,
        jobOrder.partner_id,
        jobOrder.job_order_details,
        jobOrder.job_due_date,
        jobOrder.job_priority,
        jobOrder.job_order_status,
        currentDate,
        currentDate,
        jobOrder.updated_by || null
      ];
  
      const insertResult = await this.db.run(insertQuery, insertValues);
      const newJobId = insertResult.changes?.lastId || 0;
  
      if (newJobId > 0) {
        await this.logSyncChange('botiq_job_order', 'job_id', newJobId);
        console.log(' New job order inserted with ID:', newJobId);
        return newJobId;
      } else {
        console.error(' Failed to insert job order.');
        return null;
      }
  
    } catch (error) {
      console.error(' Error saving job order:', error);
      return null;
    }
  }

        console.error(`Missing field in jobDoc: ${field}`);
        return 0;
      }
    }
  
    try {
      if (jobDoc.job_doc_id) {
        //  Update existing record
        const updateQuery = `
          UPDATE botiq_job_docs
          SET job_id = ?, 
          order_id = ?, 
          org_id = ?, 
          details_id = ?, 
          details_type = ?, 
          details_data = ?,

        const updateValues = [

        return jobDoc.job_doc_id;
      } else {
        // Insert new record
