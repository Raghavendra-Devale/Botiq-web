SELECT
      SUM(CASE WHEN due_date < date('now', 'start of month') THEN 1 ELSE 0 END) AS overdue,
      SUM(CASE WHEN due_date >= date('now', 'start of month')
               AND due_date < date('now', 'start of month', '+1 month') THEN 1 ELSE 0 END) AS m1,
      SUM(CASE WHEN due_date >= date('now', 'start of month', '+1 month')
               AND due_date < date('now', 'start of month', '+2 month') THEN 1 ELSE 0 END) AS m2,
      SUM(CASE WHEN due_date >= date('now', 'start of month', '+2 month')
               AND due_date < date('now', 'start of month', '+3 month') THEN 1 ELSE 0 END) AS m3,
      SUM(CASE WHEN due_date >= date('now', 'start of month', '+3 month') THEN 1 ELSE 0 END) AS later,
      COUNT(*) AS total
    FROM botiq_order
    WHERE org_id = ? AND order_status NOT IN ('Delivered', 'Hold')
	
// getOrderDashboardSummary for customer orders this week

SELECT 
      SUM(CASE WHEN order_status not in ('Delivered', 'Hold') AND due_date <  date('now', 'weekday 0', '-6 days') THEN 1 ELSE 0 END) AS over_due,
      SUM(CASE WHEN order_status not in ('Delivered', 'Hold') AND due_date >= date('now', 'weekday 0', '-6 days') AND due_date <=  date('now', 'weekday 0', '0 days') THEN 1 ELSE 0 END) AS due,
      SUM(CASE WHEN order_status not in ('Delivered', 'Hold') AND order_priority = 1 THEN 1 ELSE 0 END) AS urgent,
      SUM(CASE WHEN order_status = 'Delivered' and delivered_date >= date('now','weekday 0', '-6 days' ) THEN 1 ELSE 0 END) AS delivered,
      SUM(CASE WHEN order_status = 'Ready' THEN 1 ELSE 0 END) AS ready,
      SUM(CASE WHEN order_date >= date('now',  'weekday 0', '-6 days') THEN 1 ELSE 0 END) AS newthisweek
    FROM botiq_order
    WHERE org_id = ?
      AND (
        (due_date < date('now', 'weekday 0', '0 days') AND order_status not in ('Delivered', 'Hold'))
        OR order_status = 'Ready'
        OR (order_status = 'Delivered' AND delivered_date >= date('now', 'weekday 0', '-6 days'))
        OR order_date >= date('now', 'weekday 0', '-6 days' )
         or order_priority  = 1  and order_status not in ('Delivered','Hold')
)
;

  CREATE OR REPLACE FUNCTION public.get_full_dashboard(p_org_id integer)
RETURNS jsonb
LANGUAGE plpgsql
AS $$
DECLARE
  result jsonb;
BEGIN

  SELECT jsonb_build_object(

    -- 🔹 MONTHLY SUMMARY
    'monthly_due_summary', (
      SELECT jsonb_build_object(
        'overdue', SUM(CASE WHEN due_date < date_trunc('month', CURRENT_DATE) THEN 1 ELSE 0 END),
        'm1', SUM(CASE WHEN due_date >= date_trunc('month', CURRENT_DATE)
                       AND due_date < date_trunc('month', CURRENT_DATE) + INTERVAL '1 month' THEN 1 ELSE 0 END),
        'm2', SUM(CASE WHEN due_date >= date_trunc('month', CURRENT_DATE) + INTERVAL '1 month'
                       AND due_date < date_trunc('month', CURRENT_DATE) + INTERVAL '2 month' THEN 1 ELSE 0 END),
        'm3', SUM(CASE WHEN due_date >= date_trunc('month', CURRENT_DATE) + INTERVAL '2 month'
                       AND due_date < date_trunc('month', CURRENT_DATE) + INTERVAL '3 month' THEN 1 ELSE 0 END),
        'later', SUM(CASE WHEN due_date >= date_trunc('month', CURRENT_DATE) + INTERVAL '3 month' THEN 1 ELSE 0 END),
        'total', COUNT(*)
      )
      FROM botiq_order
      WHERE org_id = p_org_id
        AND order_status NOT IN ('Delivered', 'Hold')
    ),

    -- 🔹 ORDER SUMMARY
    'order_summary', (
      SELECT jsonb_build_object(
        'over_due', SUM(CASE WHEN order_status NOT IN ('Delivered','Hold')
          AND due_date < date_trunc('week', CURRENT_DATE) THEN 1 ELSE 0 END),

        'due', SUM(CASE WHEN order_status NOT IN ('Delivered','Hold')
          AND due_date >= date_trunc('week', CURRENT_DATE)
          AND due_date < date_trunc('week', CURRENT_DATE) + INTERVAL '7 days' THEN 1 ELSE 0 END),

        'urgent', SUM(CASE WHEN order_status NOT IN ('Delivered','Hold')
          AND order_priority = 1 THEN 1 ELSE 0 END),

        'delivered', SUM(CASE WHEN order_status = 'Delivered'
          AND delivered_date >= date_trunc('week', CURRENT_DATE) THEN 1 ELSE 0 END),

        'ready', SUM(CASE WHEN order_status = 'Ready' THEN 1 ELSE 0 END),

        'new_this_week', SUM(CASE WHEN order_date >= date_trunc('week', CURRENT_DATE) THEN 1 ELSE 0 END)
      )
      FROM botiq_order
      WHERE org_id = p_org_id
    ),

    -- 🔹 DUE ORDER SUMMARY
    'due_order_summary', (
      WITH expanded AS (
        SELECT
          o.order_id,
          o.due_date,
          o.order_amount,
          (j.value->>'quantity')::int AS quantity
        FROM botiq_order o
        JOIN jsonb_array_elements(o.order_details::jsonb) j ON true
        WHERE o.org_id = p_org_id
          AND o.order_status NOT IN ('Ready','Delivered','Hold')
      ),
      per_order AS (
        SELECT
          order_id,
          due_date,
          order_amount,
          SUM(quantity) AS total_quantity
        FROM expanded
        GROUP BY order_id, due_date, order_amount
      )
      SELECT jsonb_build_object(
        'due_this_week', jsonb_build_object(
          'count', SUM(CASE WHEN due_date >= date_trunc('week', CURRENT_DATE)
                            AND due_date < date_trunc('week', CURRENT_DATE) + INTERVAL '7 days'
                            THEN 1 ELSE 0 END),
          'amount', SUM(CASE WHEN due_date >= date_trunc('week', CURRENT_DATE)
                             AND due_date < date_trunc('week', CURRENT_DATE) + INTERVAL '7 days'
                             THEN order_amount ELSE 0 END),
          'quantity', SUM(CASE WHEN due_date >= date_trunc('week', CURRENT_DATE)
                               AND due_date < date_trunc('week', CURRENT_DATE) + INTERVAL '7 days'
                               THEN total_quantity ELSE 0 END)
        ),

        'overdue', jsonb_build_object(
          'count', SUM(CASE WHEN due_date < date_trunc('week', CURRENT_DATE) THEN 1 ELSE 0 END),
          'amount', SUM(CASE WHEN due_date < date_trunc('week', CURRENT_DATE) THEN order_amount ELSE 0 END),
          'quantity', SUM(CASE WHEN due_date < date_trunc('week', CURRENT_DATE) THEN total_quantity ELSE 0 END)
        )
      )
      FROM per_order
    )

  )
  INTO result;

  RETURN result;

END;
$$;
